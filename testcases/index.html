Print help.

function (test) {
    var forked = childProcess.fork(fur, ['-h'], {
        silent: true
    });
    forked.on('error', function () {
        test.ifError(new Error('Command failed.'));
    });
    forked.on('close', function (code) {
        test.equal(code, 0);
        test.done();
    });
}
